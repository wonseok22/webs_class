1.1 for 문을 이용해서 테이블을 출력하시오 !

{
    var table = "<table border='1'>";
        for (var i=0; i<5; i++){
            table += "<tr>";
            for (var j=0; j<5; j++){
                table += "<td>1</td>";
            }
            table += "</tr>";
        }
        table += "</table>";
        document.querySelector('.tab').innerHTML= table;
}
