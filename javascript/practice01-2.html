1. 구구단 출력하기

1.1 1단부터 9단까지 출력하시오

{
    //for문
    for(let i=1; i<10; i++){
        for(let j=1; j<10; j++){
            console.log(i + " * " + j + " = " , i*j);
        }
    }

    //while문
    let m = 1;
    while (m<10){
        let n = 1;
        while (n<10){
            console.log(m + " * " + n + " = " , m*n);
            n ++;
        }
        console.log('\n');
        m ++;
    }
}
