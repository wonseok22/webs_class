03. 다중 if문

{
    const num = 100;

    if(num == 90){
        document.write("실행되었습니다. (num==90)");
    } else if (num == 100){
        document.write("실행되었습니다. (num==100)");
    } else if (num == 110){
        document.write("실행되었습니다. (num==110)");
    } else if (num == 120){
        document.write("실행되었습니다. (num==120)");
    } else {
        document.write("실행되었습니다. (num==값이없음)");
    }
}

04. 중첩 if문

{
    const num = 100;

    if (num==100){
        document.write("실행되었습니다.(1)");
        if (num==100) {
            document.write("실행되었습니다.(2)");
            if (num==100){
                document.write("실행되었습니다.(3)");
            }
        }
    }
    else {
        document.write("실행되었습니다.(4)");
    }
}

06. switch

{
    const num = 100;
                
    switch(num){
        case 100 :
            document.write("실행되었습니다.(num==100)");
            break;
        case 200 :
            document.write("실행되었습니다.(num==200)");
            break;
        case 300 :
            document.write("실행되었습니다.(num==300)");
            break;
        default :
            document.write("실행되었습니다.(num==값이없음)");
            break;
    }
}

07. while문

{
    let num = 1;

    while (num <= 5){
        document.write(num + ". 실행되었습니다.","
");
        num ++;
    }
}

08. do while문

{
    let num = 1;

    do {
        document.write(num + ". 실행되었습니다.", "
");
        num ++;
    } while (num <= 5);
}

11. break문

{
    let num = 0;

    while (num <= 20){
        document.write(num);
        num ++;
        if (num == 10){
            break;
        }
    }

    for( let i = 0; i <= 20; i++){
        document.write(i);
        if(i==10){
            break;
        }
    }
}

12. continue

{
    let num = 1;
                
    //while문을 이용한 1~10 출력
    while (num<=10){
        num ++;
        if(num == 5) continue;
        document.write(num);
        
    }

    //for문을 이용한 1~10 출력
    for(let i=1; i<=10; i++){
        document.write(i);
        if (i==5) continue;
    }
}
