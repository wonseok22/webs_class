01. 변수 : 데이터 불러오기

{
    let x = 100;
    let y = 200;
    let z = "javascript";

    /* 
        let의 중복을 제거하면 
        let x = 100, y = 200, z = "javascript";
    
        또는

        let x = 100,
            y = 200,
            z = "javascript";
    */

    document.write("*** 01. 변수 ***");
    document.write(x);
    document.write(y,;
    document.write(z);
}

02. 상수 : 데이터 불러오기

{
    const x = 100,
          y = 200,
          z = "javascript";

    document.write("*** 02. 상수 ***");
    document.write(x);
    document.write(y);
    document.write(z);
}

03. 배열 : 데이터 불러오기

{
    const arr = [100,200,"javascript"];

    document.write("*** 03. 배열 ***");
    document.write(arr[0]);
    document.write(arr[1]);
    document.write(arr[2]);
}

04. 배열 : 데이터 불러오기 : 2차 배열

{
    const arr = [100,200,["javascript","jquery"]];

    document.write("*** 04. 배열 : 2차 배열 ***");
    document.write(arr[0]);
    document.write(arr[1]);
    document.write(arr[2][0]);
    document.write(arr[2][1]);
}

05. 배열 : 데이터 불러오기 : 갯수 구하기

{
    const arr = [100,200,"javascript"];

    document.write("*** 05. 배열 : 갯수 구하기 ***");
    document.write(arr.length);
}
